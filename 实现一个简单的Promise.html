# 简单实现Promise

# 构造函数

首先实现创建一个myPromise类，然后定义实例变量status, value, reason

class myPromise {
  constructor() {
        this.status = 'PENDING';
        this.value = null;
        this.reason = null;
        // 存储回调函数
        this.onFulfilledCallbacks = [];
        this.onRejectedCallbacks = [];
        // bind
        this.resolve = this.resolve.bind(this);
        this.reject = this.reject.bind(this);
        this.then = this.then.bind(this);
        // 执行传进来的fn，并传入resolve, reject函数
        try {
            fn(this.resolve, this.reject);
        } catch (error) {
            this.reject(error);
        }
  }
}

# 内部两个实例方法resolve,reject

这两个实例方法是为了改变当前的myPromise的状态，并将后面注册进去的回调队列进行遍历执行。

    resolve(value) {
        if (this.status === 'PENDING') {
            this.status = 'FULFILLED';
            this.value = value;
            this.onFulfilledCallbacks.forEach(fn => fn(value))
        }
    }

    reject(reason) {
        if (this.status === 'PENDING') {
            this.status = 'REJECTED';
            this.reason = reason;
            this.onRejectedCallbacks.forEach(fn => fn(reason))
        }
    }

# 最关键的then方法

Promise对象有一个then方法，用来注册Promise对象状态确定后的回调。

	then(onFulfilled, onRejected) {
				const self = this; // 存储this
        // onFulfilled, onRejected都应该是回调函数，不是函数类型的时候做下处理
        if (typeof onFulfilled !== 'function') {
            onFulfilled = function (value) {
                return value
            }
        }
        if (typeof onRejected !== 'function') {
            onRejected = function (value) {
                return value
            }
        }
    		// 针对三种情况下做处理
        // Promise对象存在以下三种状态，对三种状态采用不同处理
        if (self.status === 'resolved') {
            return new Promise(function(resolve, reject) {
                // todo
            });
        }
    
        if (self.status === 'rejected') {
            return new Promise(function(resolve, reject) {
                // todo
            });
        }
    
        if (self.status === 'pending') {
            return new Promise(function(resolve, reject) {
                // todo
            });
        }
	}

下面来看看针对上面代码中的三种状态应该做些什么:

  • 对于已经是完成态
        // 已经是完成态，执行onFulfilled函数，并返回promise对象

                        if (ret instanceof myPromise) {
                            ret.then(resolve, reject)

                            resolve(ret);
                        }

                        reject(error);
                    }
                })
            });
        }
  • 对于已经是拒绝态，和上面基本一样

# 最终代码

class myPromise {
    constructor(fn) {
        this.status = 'PENDING';
        this.value = null;
        this.reason = null;
        // 存储回调函数
        this.onFulfilledCallbacks = [];
        this.onRejectedCallbacks = [];
        // bind
        this.resolve = this.resolve.bind(this);
        this.reject = this.reject.bind(this);
        this.then = this.then.bind(this);
        // 执行传进来的fn，并传入resolve, reject函数
        try {
            fn(this.resolve, this.reject);
        } catch (error) {
            this.reject(error);
        }
    }

    resolve(value) {
        if (this.status === 'PENDING') {
            this.status = 'FULFILLED';
            this.value = value;
            this.onFulfilledCallbacks.forEach(fn => fn(value))
        }
    }

    reject(reason) {
        if (this.status === 'PENDING') {
            this.status = 'REJECTED';
            this.reason = reason;
            this.onRejectedCallbacks.forEach(fn => fn(reason))
        }
    }

    then(onFulfilled, onRejected) {
        const self = this; // 存储this
        // 不是函数的时候做下处理
        if (typeof onFulfilled !== 'function') {
            onFulfilled = function (value) {
                return value
            }
        }
        if (typeof onRejected !== 'function') {
            onRejected = function (value) {
                return value
            }
        }
        // 对三种情况的处理
        // 已经是完成态，执行onFulfilled函数，并返回promise对象
        if (self.status === 'FULFILLED') {
            return new myPromise((resolve, reject) => {
                // 规范要求是异步的
                setTimeout(() => {
                    try {
                        const ret = onFulfilled(self.value); // ret是onFulfilled的返回值
                        // 如果ret是一个promise，将resolve,reject传进去作为onFulfilled，onRejected，取其resolve的值作为新的promise的结果
                        if (ret instanceof myPromise) {
                            ret.then(resolve, reject)
                        } else {
                            resolve(ret);
                        }
                    } catch (error) {
                        reject(error);
                    }
                })
            });
        }
        // 如果是拒绝态，和上面类似
        if (self.status === 'REJECTED') {
            return new myPromise((resolve, reject) => {
                setTimeout(() => {
                    try {
                        const ret = onRejected(self.reason); // ret是onFulfilled的返回值
                        // 如果ret是一个promise，将resolve,reject传进去作为onFulfilled，onRejected，取其resolve的值作为新的promise的结果
                        if (ret instanceof myPromise) {
                            ret.then(resolve, reject)
                        } else {
                            reject(ret);
                        }
                    } catch (error) {
                        reject(error);
                    }
                });
            });
        }
        // 如果还是PENDING状态，将回调注册到相应的回调队列上，并返回promise对象
        if (self.status === 'PENDING') {
            return new myPromise((resolve, reject) => {
                self.onFulfilledCallbacks.push(() => {
                    setTimeout(() => {
                        try {
                            const ret = onFulfilled(self.value); // ret是onFulfilled的返回值
                            // 如果ret是一个promise，将resolve,reject传进去作为onFulfilled，onRejected，取其resolve的值作为新的promise的结果
                            if (ret instanceof myPromise) {
                                ret.then(resolve, reject)
                            } else {
                                resolve(ret);
                            }
                        } catch (error) {
                            reject(error);
                        }
                    })
                });
                self.onRejectedCallbacks.push(() => {
                    setTimeout(() => {
                        try {
                            const ret = onRejected(self.reason); // ret是onFulfilled的返回值
                            // 如果ret是一个promise，将resolve,reject传进去作为onFulfilled，onRejected，取其resolve的值作为新的promise的结果
                            if (ret instanceof myPromise) {
                                ret.then(resolve, reject)
                            } else {
                                reject(ret);
                            }
                        } catch (error) {
                            reject(error);
                        }
                    });
                });
            });
        }
    }
}
